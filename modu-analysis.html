  • 예약 결제 직전, 일부 사용자에게 "당일 자정까지 반납 시간을 연장하는" 옵션을 정액 4,900원에 추천하는 실험

  • 정상 예약 가격 대비 큰 할인이지만 어차피 비어있던 시간이라 회사 입장 기회비용 거의 없음

  • B군 (실험군): 업셀링 바텀시트 노출. 시간연장 옵션 선택 가능
  • 전체 예약 회원을 50:50으로 무작위 분배
  • B군 안에서 9개 노출 조건을 모두 만족하는 회원만 실제로 업셀링 바텀시트을 봄

  • BigQuery: 소카 내부 데이터 조회 시스템. 본 분석의 수치 소스

5-1. 모수 적합성 판단 기준

본 실험은 통계적 유의성 검정보다 비즈니스 판단 (시간연장 옵션이 유휴 시간 채우는 도구로 의미 있는가) 이 목적

  • 수락률 추정 정확도
    • 옵션 노출 회원이 약 380명 이상 누적되면 "수락률 약 33% (±5%p)" 안정적 추정 가능
    • 현재 (2차): 5/19~5/26 약 317명, 6/2 종료 시 누적 약 500~600명 예상
  • 일별 변동성 보정
    • 일평균 노출 50명 이상이면 일별 수락 패턴 (요일별 차이 등) 관찰 가능
    • 현재 일평균 약 60명이라 보정 가능 수준
  • 유휴 시간 활용 실적 누적
    • 월 누적 추가 사용된 차량 시간이 5,000시간 이상이면 회사 차원에서 "가치 있다" 판단 가능 (기준값은 부서 결정)
    • 현재 약 4,030시간/월이라 임계점 부근

노출 조건 단계별 모수

WITH base AS (
  SELECT DATE(created_at, 'Asia/Seoul') AS create_dt,
    DATETIME(start_at, 'Asia/Seoul') AS start_kst,
    DATETIME(end_at, 'Asia/Seoul') AS end_kst,
    DATETIME(created_at, 'Asia/Seoul') AS create_kst,
    way, cancel_at
  FROM `socar-data.tianjin_replica.reservation_info`
  WHERE DATE(created_at, 'Asia/Seoul') BETWEEN '2026-05-19' AND '2026-05-21'
)
SELECT COUNT(*) AS step0_all,
  COUNT(CASE WHEN way='round' AND cancel_at IS NULL THEN 1 END) AS step1_round,
  COUNT(CASE WHEN way='round' AND cancel_at IS NULL
         AND DATETIME_DIFF(start_kst, create_kst, HOUR) <= 24 THEN 1 END) AS step2_lead24h,
  COUNT(CASE WHEN way='round' AND cancel_at IS NULL
         AND DATETIME_DIFF(start_kst, create_kst, HOUR) <= 24
         AND EXTRACT(HOUR FROM start_kst) BETWEEN 9 AND 14
         AND EXTRACT(HOUR FROM end_kst) BETWEEN 15 AND 19
         AND DATETIME_DIFF(end_kst, start_kst, HOUR) BETWEEN 4 AND 24 THEN 1 END) AS step3_time
FROM base

객단가 평균/표준편차 (검정력 계산용)

WITH target AS (
  SELECT ri.id AS rsv_id
  FROM `socar-data.tianjin_replica.reservation_info` ri
  WHERE DATE(ri.created_at, 'Asia/Seoul') BETWEEN '2026-05-01' AND '2026-05-15'
    AND ri.way = 'round' AND ri.cancel_at IS NULL
    AND DATETIME_DIFF(DATETIME(ri.start_at,'Asia/Seoul'),
                      DATETIME(ri.created_at,'Asia/Seoul'), HOUR) <= 24
    AND EXTRACT(HOUR FROM DATETIME(ri.start_at,'Asia/Seoul')) BETWEEN 9 AND 14
    AND EXTRACT(HOUR FROM DATETIME(ri.end_at,'Asia/Seoul')) BETWEEN 15 AND 19
)
SELECT COUNT(*) AS n,
       AVG(p.revenue) AS mean_rev,
       STDDEV(p.revenue) AS std_rev
FROM target t
JOIN `socar-data.socar_biz_profit.profit_socar_reservation` p
  ON t.rsv_id = p.reservation_id
