검증용 BigQuery 쿼리
1. 노출 일별 (웹 이벤트, ut=i)
SELECT event_date_kst AS dt, COUNT(DISTINCT IF(event_name='view', member_id, NULL)) AS view_m, COUNT(DISTINCT IF(component_name='interval_option', member_id, NULL)) AS optclick_m FROM `socar-data.app_web_log.socar_app_web_log` WHERE page_name='upsell_bottomsheet' AND event_date_kst BETWEEN '2026-06-25' AND '2026-06-28' AND REGEXP_EXTRACT(form_data, r'ut\\?:\\?([a-zA-Z_]+)') = 'i' GROUP BY dt ORDER BY dt -- 결과: 6/25 72, 6/26 154, 6/27 74, 6/28 228 (누계 distinct 521)
2. 수락 구간별 (결제 meta content_token 디코딩)
-- content_token(JWT)의 payload를 base64 디코딩해 customerSurcharge로 구간 구분 WITH te AS ( SELECT pi.reservation_id, ri.cancel_at, SAFE_CAST(JSON_VALUE(SAFE.PARSE_JSON(SAFE_CONVERT_BYTES_TO_STRING(FROM_BASE64( REPLACE(REPLACE(SPLIT(JSON_VALUE(pmi.meta,'$.content_token'),'.')[OFFSET(1)],'-','+'),'_','/') ))),'$.data') AS STRING) AS d FROM `socar-data.tianjin_replica.paid_meta_info` pmi JOIN `socar-data.tianjin_replica.paid_info` pi ON pi.id = pmi.id JOIN `socar-data.tianjin_replica.reservation_info` ri ON ri.id = pi.reservation_id WHERE REGEXP_CONTAINS(pmi.meta, r'TIME_EXTENSION') AND DATE(pmi.created_at,'Asia/Seoul') BETWEEN '2026-06-25' AND '2026-06-29') SELECT SAFE_CAST(JSON_VALUE(SAFE.PARSE_JSON(d),'$.customerSurcharge') AS INT64) AS surcharge, COUNT(DISTINCT reservation_id) AS accept_resv, COUNT(DISTINCT IF(cancel_at IS NULL, reservation_id, NULL)) AS active_resv FROM te GROUP BY surcharge ORDER BY surcharge -- 결과: 4900 -> 수락 161 / 순 125, 12900 -> 수락 9 / 순 7
3. 구간(가격)과 반납시간 일치 검증
-- JWT payload의 originalInterval.end(원래 예약 반납시각)를 KST 시로 변환해 구간과 대조 -- te는 위 2번과 동일 SELECT SAFE_CAST(JSON_VALUE(SAFE.PARSE_JSON(d),'$.customerSurcharge') AS INT64) AS surcharge, EXTRACT(HOUR FROM DATETIME(TIMESTAMP( JSON_VALUE(SAFE.PARSE_JSON(d),'$.payload.originalInterval.end')),'Asia/Seoul')) AS return_hour, COUNT(DISTINCT reservation_id) AS resv FROM te GROUP BY surcharge, return_hour ORDER BY surcharge, return_hour -- 결과: 4900 -> 반납 15~20시만, 12900 -> 반납 12~14시만 (가격 로직 정상)
