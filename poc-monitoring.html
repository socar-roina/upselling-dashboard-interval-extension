검증용 BigQuery 쿼리
1. 노출 일별 (웹 이벤트, ut=i)
SELECT event_date_kst AS dt, COUNT(DISTINCT IF(event_name='view', member_id, NULL)) AS view_m, COUNT(DISTINCT IF(component_name='interval_option', member_id, NULL)) AS optclick_m FROM `socar-data.app_web_log.socar_app_web_log` WHERE page_name='upsell_bottomsheet' AND event_date_kst BETWEEN '2026-06-25' AND '2026-07-04' AND REGEXP_EXTRACT(form_data, r'ut\\??:\\??([a-z]+)') = 'i' GROUP BY dt ORDER BY dt -- 노출(view_m): 54,125,66,212,209,216,216,269,172,74 (누계 distinct 1544) -- 옵션클릭(optclick_m): 19,52,32,100,68,73,71,87,67,23 (누계 distinct 587)
2. 수락 구간별 (결제 meta content_token 디코딩)
-- content_token(JWT)의 payload를 base64 디코딩해 customerSurcharge로 구간 구분 WITH te AS ( SELECT pi.reservation_id, ri.cancel_at, SAFE_CAST(JSON_VALUE(SAFE.PARSE_JSON(SAFE_CONVERT_BYTES_TO_STRING(FROM_BASE64( REPLACE(REPLACE(SPLIT(JSON_VALUE(pmi.meta,'$.content_token'),'.')[OFFSET(1)],'-','+'),'_','/') ))),'$.data') AS STRING) AS d FROM `socar-data.tianjin_replica.paid_meta_info` pmi JOIN `socar-data.tianjin_replica.paid_info` pi ON pi.id = pmi.id JOIN `socar-data.tianjin_replica.reservation_info` ri ON ri.id = pi.reservation_id WHERE REGEXP_CONTAINS(pmi.meta, r'TIME_EXTENSION') AND DATE(pmi.created_at,'Asia/Seoul') BETWEEN '2026-06-25' AND '2026-07-04') SELECT SAFE_CAST(JSON_VALUE(SAFE.PARSE_JSON(d),'$.customerSurcharge') AS INT64) AS surcharge, COUNT(DISTINCT reservation_id) AS accept_resv, COUNT(DISTINCT IF(cancel_at IS NULL, reservation_id, NULL)) AS active_resv FROM te GROUP BY surcharge ORDER BY surcharge -- 결과: 4900 -> 수락 398 / 순 315, 12900 -> 수락 20 / 순 15
3. 구간(가격)과 반납시간 일치 검증
-- JWT payload의 originalInterval.end(원래 예약 반납시각)를 KST 시로 변환해 구간과 대조 -- te는 위 2번과 동일 SELECT SAFE_CAST(JSON_VALUE(SAFE.PARSE_JSON(d),'$.customerSurcharge') AS INT64) AS surcharge, EXTRACT(HOUR FROM DATETIME(TIMESTAMP( JSON_VALUE(SAFE.PARSE_JSON(d),'$.payload.originalInterval.end')),'Asia/Seoul')) AS return_hour, COUNT(DISTINCT reservation_id) AS resv FROM te GROUP BY surcharge, return_hour ORDER BY surcharge, return_hour -- 결과: 4900 -> 반납 15~20시만, 12900 -> 반납 12~14시만 (가격 로직 정상)
4. 전체뷰 대비 시간연장뷰 침투율 (완화 전후)
WITH daily AS ( SELECT DATE(event_at_kst) AS dt, COUNT(DISTINCT IF(event_name='view', member_id, NULL)) AS total_view, COUNT(DISTINCT IF(event_name='view' AND (CONTAINS_SUBSTR(form_data,'intervalUpselling') OR REGEXP_EXTRACT(form_data, r'ut\\??:\\??([a-z]+)')='i'), member_id, NULL)) AS interval_view FROM `socar-data.app_web_log.socar_app_web_log` WHERE page_name='upsell_bottomsheet' AND DATE(event_at_kst) BETWEEN '2026-05-21' AND '2026-07-01' GROUP BY dt ) SELECT CASE WHEN dt<='2026-06-02' THEN '완화 전' WHEN dt>='2026-06-25' THEN '완화 후' END AS period, ROUND(AVG(total_view)) AS total_view_avg, ROUND(AVG(interval_view),1) AS interval_view_avg, ROUND(SUM(interval_view)/SUM(total_view)*100,2) AS penetration_pct FROM daily WHERE dt<='2026-06-02' OR dt>='2026-06-25' GROUP BY period ORDER BY period -- 결과: 완화 전 7468 / 74 / 1.0%, 완화 후 8468 / 170 / 2.0% (6/25~7/2) -- 주의: CONTAINS_SUBSTR('ut:i')는 이스케이프 인코딩(ut\:\i) 누락. 정규식으로 잡아야 함
5. 반납 요일별 수락 (주말 수요 확인)
-- te는 위 2번과 동일 (결제 meta TIME_EXTENSION, 6/25~7/1) SELECT FORMAT_DATETIME('%a', DATETIME(TIMESTAMP( JSON_VALUE(SAFE.PARSE_JSON(d),'$.payload.originalInterval.end')),'Asia/Seoul')) AS return_dow, COUNT(DISTINCT reservation_id) AS accept, COUNT(DISTINCT IF(cancel_at IS NULL, reservation_id, NULL)) AS active FROM te GROUP BY return_dow ORDER BY active DESC -- 결과(취소 제외 수락): 일 58, 화 43, 수 42, 금 41, 월 28, 목 14
6. 수락자의 원래 반납 시각 분포
-- 각 수락자가 업셀링 제안 전 원래 예약했던 반납 시각(KST) -- te는 위 2번과 동일 (결제 meta TIME_EXTENSION, 6/25~7/4) -- originalInterval.end = 연장 전 원래 반납 시각. content_token(JWT) payload에서 디코딩 SELECT SAFE_CAST(JSON_VALUE(SAFE.PARSE_JSON(d),'$.customerSurcharge') AS INT64) AS surcharge, EXTRACT(HOUR FROM DATETIME(TIMESTAMP( JSON_VALUE(SAFE.PARSE_JSON(d),'$.payload.originalInterval.end')),'Asia/Seoul')) AS orig_return_hour, COUNT(DISTINCT IF(cancel_at IS NULL, reservation_id, NULL)) AS active FROM te GROUP BY surcharge, orig_return_hour ORDER BY surcharge, orig_return_hour -- 결과(취소 제외): 12900 -> 12시 5, 13시 4, 14시 6 -- 4900 -> 15시 31, 16시 33, 17시 46, 18시 76, 19시 74, 20시 55 -- 합계 330건. 원래 반납이 15-20시면 4,900원, 12-14시면 12,900원 구간
7. 유저 최초 설정 이용시간 vs 수락 후 (건별)
-- 수락한 예약별로 "유저가 최초 설정한 이용 구간"과 "연장 후 반납"을 나란히 조회 -- te는 위 2번과 동일 (결제 meta TIME_EXTENSION, 6/25~7/2) -- originalInterval = 업셀 제안 전 원래 예약 구간, extendedInterval = 연장 후 구간 SELECT reservation_id, DATETIME(TIMESTAMP(JSON_VALUE(SAFE.PARSE_JSON(d),'$.payload.originalInterval.start')),'Asia/Seoul') AS orig_start, -- 최초 대여 시작 DATETIME(TIMESTAMP(JSON_VALUE(SAFE.PARSE_JSON(d),'$.payload.originalInterval.end')),'Asia/Seoul') AS orig_end, -- 최초 반납 DATETIME(TIMESTAMP(JSON_VALUE(SAFE.PARSE_JSON(d),'$.payload.extendedInterval.end')),'Asia/Seoul') AS extended_end, -- 연장 후 반납 SAFE_CAST(JSON_VALUE(SAFE.PARSE_JSON(d),'$.customerSurcharge') AS INT64) AS surcharge FROM te WHERE cancel_at IS NULL ORDER BY reservation_id DESC -- 예: 최초 11:20~20:00 반납 -> 연장 후 익일 00:00, +4,900 / 최초 08:30~13:30 -> 익일 00:00, +12,900
8. (업셀 공통) 차종 수락 건별 최초/업그레이드 차종
-- 차종 업그레이드 수락 건. 결제 JWT CAR_UPGRADE payload에 원본·제안 차종이 함께 있음 -- (시간연장과 동일 구조. 차종 ABT 운영기에만 데이터 있음) WITH cu AS ( SELECT pi.reservation_id, ri.cancel_at, SAFE_CAST(JSON_VALUE(SAFE.PARSE_JSON(SAFE_CONVERT_BYTES_TO_STRING(FROM_BASE64( REPLACE(REPLACE(SPLIT(JSON_VALUE(pmi.meta,'$.content_token'),'.')[OFFSET(1)],'-','+'),'_','/') ))),'$.data') AS STRING) AS d FROM `socar-data.tianjin_replica.paid_meta_info` pmi JOIN `socar-data.tianjin_replica.paid_info` pi ON pi.id = pmi.id JOIN `socar-data.tianjin_replica.reservation_info` ri ON ri.id = pi.reservation_id WHERE REGEXP_CONTAINS(pmi.meta, r'CAR_UPGRADE') AND DATE(pmi.created_at,'Asia/Seoul') BETWEEN '2026-06-01' AND '2026-07-02') SELECT reservation_id, JSON_VALUE(SAFE.PARSE_JSON(d),'$.payload.originalCarClassId') AS orig_car_class, -- 최초 차종 JSON_VALUE(SAFE.PARSE_JSON(d),'$.payload.upgradedCarClassId') AS upgraded_car_class, -- 업그레이드 차종 SAFE_CAST(JSON_VALUE(SAFE.PARSE_JSON(d),'$.customerSurcharge') AS INT64) AS surcharge FROM cu WHERE cancel_at IS NULL ORDER BY reservation_id DESC -- 예: 410 -> 170 (+9,900). 노출 로그로도 확인 가능: form_data의 co(원본)/cu(업그레이드)
9. (업셀 공통) upsell_type · 기존 면책(PF) — 노출 로그 REGEXP
-- form_data는 {pf:70,ut:p}처럼 따옴표 없는 비표준 문자열이라 JSON_VALUE 불가, REGEXP로 파싱 SELECT CASE REGEXP_EXTRACT(form_data, r'ut\??:\??([a-z]+)') WHEN 'i' THEN '시간연장' WHEN 'c' THEN '차종업그레이드' WHEN 'p' THEN 'PF넛지' END AS upsell_type, REGEXP_EXTRACT(form_data, r'pf:([0-9A-Za-z_]+)') AS orig_pf, -- 기존(원래) 면책: 30/70 = 30·70만원면책 REGEXP_EXTRACT(form_data, r'co:([0-9]+)') AS orig_car_class, -- 차종만 REGEXP_EXTRACT(form_data, r'cu:([0-9]+)') AS upgraded_car_class -- 차종만 FROM `socar-data.app_web_log.socar_app_web_log` WHERE page_name='upsell_bottomsheet' AND event_name='view' AND event_date_kst BETWEEN '2026-06-25' AND '2026-07-01' -- 로깅 정의서 4714070038 참조 (ut=업셀종류, pf=기존 면책, co/cu=차종 원본/업그레이드)

  • 신규 12,900원 구간은 정상 작동. 조건대로 잘 붙었고, 다만 아직 수락이 15건(취소 제외)으로 적음
