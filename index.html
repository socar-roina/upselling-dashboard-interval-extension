1. 화면별 동작별 일일 집계 (옵션 화면 본 / 시간연장 카드 누름 / 결제 버튼 누름)

3. 실제 수락 건수와 매출 (일별, 환불 제외, 시간연장만 분리)
paid_meta_info의 meta JSON에서 upselling_type='TIME_EXTENSION'만 추출해서 시간연장 결제만 카운트
WITH net_data AS ( SELECT DATE(pi.created_at, "Asia/Seoul") AS dt, pi.reservation_id, SUM(pi.amount) AS net_amount FROM `socar-data.tianjin_replica.paid_info` pi JOIN `socar-data.tianjin_replica.paid_meta_info` pm ON pm.id = pi.id WHERE DATE(pi.created_at, "Asia/Seoul") BETWEEN "2026-05-13" AND "2026-05-22" AND pi.paid_type = 'upselling_discount' AND JSON_EXTRACT_SCALAR(pm.meta, "$.upselling_type") = 'TIME_EXTENSION' GROUP BY dt, pi.reservation_id ) SELECT dt, SUM(CASE WHEN net_amount > 0 THEN 1 ELSE 0 END) AS net_accepted_rsv, SUM(CASE WHEN net_amount > 0 THEN net_amount ELSE 0 END) AS company_burden_net, SUM(CASE WHEN net_amount > 0 THEN 4900 ELSE 0 END) AS revenue_increment FROM net_data GROUP BY dt ORDER BY dt

SELECT CASE WHEN event_at_kst < DATETIME("2026-05-19 15:50:05") THEN "1st" ELSE "2nd" END AS phase, COUNT(DISTINCT CASE WHEN event_name='view' AND component_name IS NULL THEN member_id END) AS view_uu, COUNT(DISTINCT CASE WHEN event_name='click' AND component_name='interval_option' THEN member_id END) AS click_uu FROM `socar-data.app_web_log.socar_app_web_log` WHERE event_date_kst BETWEEN "2026-05-13" AND "2026-05-22" AND event_at_kst >= DATETIME("2026-05-13 14:09:30") AND page_name = 'upsell_bottomsheet' GROUP BY phase ORDER BY phase
5. 차량 이용 시간 증가량 (본 목적 카드용)
수락 예약의 최종 반납 시각에서 원래 반납 시각을 빼서 시간 차이를 합산
WITH net AS ( SELECT CASE WHEN DATETIME(created_at, "Asia/Seoul") < DATETIME("2026-05-19 15:50:05") THEN "1st" ELSE "2nd" END AS phase, DATE(created_at, "Asia/Seoul") AS dt, reservation_id, SUM(amount) AS net_amount FROM `socar-data.dbt.stg_tianjin_replica__paid_info` WHERE DATE(created_at, "Asia/Seoul") BETWEEN "2026-05-13" AND "2026-05-22" AND paid_type = 'upselling_discount' AND DATETIME(created_at, "Asia/Seoul") >= DATETIME("2026-05-13 14:09:30") GROUP BY phase, dt, reservation_id HAVING net_amount > 0 ) SELECT n.phase, COUNT(*) AS n_accepted, ROUND(AVG(TIMESTAMP_DIFF(r.reserved_end_at, r.end_at, MINUTE)/60.0), 2) AS avg_extend_h, ROUND(SUM(TIMESTAMP_DIFF(r.reserved_end_at, r.end_at, MINUTE)/60.0), 1) AS total_extend_h FROM net n JOIN `socar-data.dbt.stg_tianjin_replica__reservation_info` r ON r.id = n.reservation_id GROUP BY n.phase ORDER BY n.phase
