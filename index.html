WITH te_rsv AS ( SELECT DISTINCT pi.reservation_id FROM `socar-data.tianjin_replica.paid_info` pi JOIN `socar-data.tianjin_replica.paid_meta_info` pm ON pm.id = pi.id WHERE DATE(pi.created_at, "Asia/Seoul") BETWEEN "2026-05-13" AND "2026-05-27" AND pi.paid_type = 'upselling_discount' AND JSON_EXTRACT_SCALAR(pm.meta, "$.upselling_type") = 'TIME_EXTENSION' ), net AS ( SELECT DATE(pi.created_at, "Asia/Seoul") AS dt, pi.reservation_id, SUM(pi.amount) AS net_amount FROM `socar-data.tianjin_replica.paid_info` pi JOIN te_rsv t ON t.reservation_id = pi.reservation_id WHERE DATE(pi.created_at, "Asia/Seoul") BETWEEN "2026-05-13" AND "2026-05-27" AND pi.paid_type = 'upselling_discount' GROUP BY dt, pi.reservation_id ) SELECT dt, SUM(CASE WHEN net_amount > 0 THEN 1 ELSE 0 END) AS accepted, SUM(CASE WHEN net_amount > 0 THEN net_amount ELSE 0 END) AS burden, SUM(CASE WHEN net_amount > 0 THEN 4900 ELSE 0 END) AS revenue FROM net GROUP BY dt ORDER BY dt
4. 1차/2차 unique 노출·클릭 유저 수 (수락률 분모용)
전체 기간 unique member_id 기준. 일별 합과 다른 이유는 회원 중복 제거
WITH base AS ( SELECT member_id, CASE WHEN event_at_kst < DATETIME("2026-05-19 15:50:05") THEN "1st" ELSE "2nd" END AS phase, event_name, component_name, form_data, event_date_kst FROM `socar-data.app_web_log.socar_app_log` WHERE event_date_kst BETWEEN "2026-05-13" AND "2026-05-25" AND page_name = 'upsell_bottomsheet' AND event_at_kst >= DATETIME("2026-05-13 14:09:30") AND member_id IS NOT NULL ) SELECT phase, COUNT(DISTINCT IF(event_name='view' AND component_name IS NULL AND (event_date_kst <= '2026-05-20' OR REGEXP_CONTAINS(form_data, r'intervalUpselling')), member_id, NULL)) AS view_uu, COUNT(DISTINCT IF(event_name='click' AND component_name='interval_option', member_id, NULL)) AS click_uu FROM base GROUP BY phase ORDER BY phase
5. 1차/2차 가동 시간 증분 (본 목적 카드용, 시간연장만)
시간연장 reservation_id 식별 후, reservation_info의 reserved_end_at(연장 후 반납)에서 end_at(실제 반납)을 뺀 시간을 합산
WITH te_rsv AS ( SELECT DISTINCT pi.reservation_id FROM `socar-data.tianjin_replica.paid_info` pi JOIN `socar-data.tianjin_replica.paid_meta_info` pm ON pm.id = pi.id WHERE DATE(pi.created_at, "Asia/Seoul") BETWEEN "2026-05-13" AND "2026-05-25" AND pi.paid_type = 'upselling_discount' AND JSON_EXTRACT_SCALAR(pm.meta, "$.upselling_type") = 'TIME_EXTENSION' ), net AS ( SELECT pi.reservation_id, CASE WHEN DATETIME(MIN(pi.created_at), "Asia/Seoul") < DATETIME("2026-05-19 15:50:05") THEN "1st" ELSE "2nd" END AS phase, SUM(pi.amount) AS net_amount FROM `socar-data.tianjin_replica.paid_info` pi JOIN te_rsv t ON t.reservation_id = pi.reservation_id WHERE DATE(pi.created_at, "Asia/Seoul") BETWEEN "2026-05-13" AND "2026-05-25" AND pi.paid_type = 'upselling_discount' GROUP BY pi.reservation_id HAVING net_amount > 0 ) SELECT n.phase, COUNT(*) AS accepted, SUM(n.net_amount) AS burden, ROUND(AVG(TIMESTAMP_DIFF(r.reserved_end_at, r.end_at, MINUTE)/60.0), 2) AS avg_extend_h, ROUND(SUM(TIMESTAMP_DIFF(r.reserved_end_at, r.end_at, MINUTE)/60.0), 1) AS total_extend_h FROM net n JOIN `socar-data.dbt.stg_tianjin_replica__reservation_info` r ON r.id = n.reservation_id GROUP BY n.phase ORDER BY n.phase

WITH te_rsv AS ( SELECT DISTINCT pi.reservation_id FROM `socar-data.tianjin_replica.paid_info` pi JOIN `socar-data.tianjin_replica.paid_meta_info` pm ON pm.id = pi.id WHERE DATE(pi.created_at, "Asia/Seoul") IN ('2026-05-14','2026-05-15','2026-05-17','2026-05-18') AND pi.paid_type = 'upselling_discount' AND JSON_EXTRACT_SCALAR(pm.meta, "$.upselling_type") = 'TIME_EXTENSION' ), net AS ( SELECT pi.reservation_id, SUM(pi.amount) AS net_amount FROM `socar-data.tianjin_replica.paid_info` pi JOIN te_rsv t ON t.reservation_id = pi.reservation_id WHERE DATE(pi.created_at, "Asia/Seoul") IN ('2026-05-14','2026-05-15','2026-05-17','2026-05-18') AND pi.paid_type = 'upselling_discount' GROUP BY pi.reservation_id HAVING net_amount > 0 ) SELECT COUNT(*) AS accepted, SUM(n.net_amount) AS burden, ROUND(AVG(TIMESTAMP_DIFF(r.reserved_end_at, r.end_at, MINUTE)/60.0), 2) AS avg_extend_h, ROUND(SUM(TIMESTAMP_DIFF(r.reserved_end_at, r.end_at, MINUTE)/60.0), 1) AS total_extend_h FROM net n JOIN `socar-data.dbt.stg_tianjin_replica__reservation_info` r ON r.id = n.reservation_id
